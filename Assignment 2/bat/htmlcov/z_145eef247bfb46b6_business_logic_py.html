12def type_of_patron(age): 

13 ''' 

14 Return a string describing the type of patron based on their age. 

15 Args: 

16 age (int): the age to describe. 

17 

18 Returns: 

19 A string describing the age, according to the rules: 

20 - "ERROR" if the age is less than 0. 

21 - "Minor" if the age is 0 or more, and less than 18. 

22 - "Adult" if the age is 18 or more, and less than 90. 

23 - "Elderly" if the age is 90 or more. 

24 ''' 

25 if age < 0: 

26 return "ERROR" 

27 elif age < 18: 

28 return "Minor" 

29 elif age < 90: 

30 return "Adult" 

31 else: 

32 return "Elderly"

35def can_borrow(type_of_item, patron_age, length_of_loan, outstanding_fees, gardening_tool_training, carpentry_tool_training): 

36 ''' 

37 Determine whether a loan can occur. 

38 Args: 

39 type_of_item (string): either "Book", "Gardening tool", or "Carpentry tool". 

40 patron_age (int): the age of the patron wanting to borrow the item, in years. 

41 length_of_loan (int): the number of days the patron wants to loan the item for. 

42 outstanding_fees (float): the fees the patron owes, before any discounts are considered. 

43 gardening_tool_training (bool): whether the patron has completed the gardening tool training or not. 

44 carpentry_tool_training (bool): whether the patron has completed the carpentry tool training or not. 

45 Returns: 

46 True if the patron is allowed to borrow the item, otherwise false. False if an invalid item type is provided. 

47 ''' 

48 if type_of_item == "Book": 

49 return can_borrow_book(patron_age, length_of_loan, outstanding_fees) 

50 elif type_of_item == "Gardening tool": 

51 return can_borrow_gardening_tool(patron_age, length_of_loan, outstanding_fees, gardening_tool_training) 

52 elif type_of_item == "Carpentry tool": 

53 return can_borrow_carpentry_tool(patron_age, length_of_loan, outstanding_fees, carpentry_tool_training) 

54 

55 return False

58def can_borrow_book(patron_age, length_of_loan, outstanding_fees): 

59 ''' 

60 Determine whether a book loan can occur. 

61 Args: 

62 patron_age (int): the age of the patron wanting to borrow the item, in years. 

63 length_of_loan (int): the number of days the patron wants to loan the item for. 

64 outstanding_fees (float): the fees the patron owes, before any discounts are considered. 

65 Returns: 

66 True if the patron is allowed to borrow the book, otherwise false. 

67 A patron is allowed to borrow a book for up to 8 weeks if they have no fees owed (i.e., fees 

68 to pay after discounts are considered). If the patron has fees owed they are not allowed 

69 to borrow a book. 

70 ''' 

71 if length_of_loan >= 56: 

72 return False 

73 

74 discount = calculate_discount(patron_age) 

75 fees_owed = outstanding_fees - (outstanding_fees * (discount / 100)) 

76 

77 if fees_owed > 0: 

78 return False 

79 

80 return True

167def calculate_discount(age): 

168 ''' 

169 Calculate the discount a patron is entitled to based on their age. 

170 Args: 

171 age (int): a patron's age in years. 

172 Returns: 

173 The discount the patron is entitled to, as a whole number percentage, or "ERROR" 

174 if an invalid (i.e., negative) age is entered. 

175 - Patrons under 50 receive no discount. 

176 - Patrons aged 50 and over, but under 65 receive a 10% discount. 

177 - Patrons aged 65 and over, but under 90 receive a 15% discount. 

178 - Patrons aged 90 and over receive a 100% discount. 

179 ''' 

180 if age < 0: 

181 return "ERROR" 

182 elif age < 50: 

183 return 0 

184 elif (age >= 50) and (age < 65): 

185 return 10 

186 elif (age >= 65) and (age < 90): 

187 return 15 

188 elif age >= 90: 

189 return 100

192def process_return(patron, item_id): 

193 ''' 

194 Process the return of an item. 

195 Removes the loan from the patron, and adjusts the "on loan" count for the item. 

196 Args: 

197 patron (Patron): the patron returning the item. It is assumed 

198 this is the patron who loaned the item. 

199 item_id (int): the ID of the item being returned. 

200 ''' 

201 to_return = patron.find_loan(item_id) 

202 to_return._item._on_loan - 1 

203 patron._loans.remove(to_return)

206def process_loan(patron, item, length_of_loan): 

207 ''' 

208 Process the loan of an item. 

209 Checks that the item can be borrowed, and if so, creates a loan with an appropriate due date, 

210 assigns the loan to the patron, and adjusts the "on loan" count for the item. 

211 Args: 

212 patron (Patron): the patron returning the item. It is assumed 

213 this the patron does not already have an active loan for the item. 

214 item (BorrowableItem): the item the patron wants to loan. 

215 length_of_loan (int): the number of days the patron wants to borrow the item for. 

216 Returns: 

217 True if the loan was successful, or false if it could not be completed. 

218 ''' 

219 due_date = date.today() + timedelta(days=length_of_loan) 

220 

221 if can_borrow(item._type, patron._age, length_of_loan, patron._outstanding_fees, patron._gardening_tool_training, patron._carpentry_tool_training): 

222 new_loan = Loan(item, due_date) 

223 patron._loans.append(new_loan) 

224 item._on_loan += 1 

225 return True 

226 else: 

227 return False
